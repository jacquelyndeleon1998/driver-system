function doPost(e) { try { const sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Allowances"); if (!e || !e.parameter || !e.parameter.data) { return jsonResponse({ success: false, message: "NO_DATA_RECEIVED" }); } // MAIN DATA const data = JSON.parse(e.parameter.data); // RECEIPTS (base64 array from frontend) const receipts = JSON.parse(e.parameter.receipts || "[]"); let receiptUrls = []; // UPLOAD RECEIPTS TO DRIVE if (receipts.length > 0) { receipts.forEach(r => { const url = uploadBase64ToDrive( r.base64, r.name, data.driver, data.tripDate ); receiptUrls.push(url); }); } // TOTAL COMPUTATION const total = Number(data.breakfast || 0) + Number(data.lunch || 0) + Number(data.dinner || 0) + Number(data.parkingFee || 0); // UNIQUE KEY FOR DUPLICATE CHECK const newKey = `${data.tripDate}_${data.driver}_${data.passenger}`; const values = sheet.getDataRange().getValues(); for (let i = 1; i < values.length; i++) { const existingKey = `${values[i][1]}_${values[i][2]}_${values[i][3]}`; if (existingKey === newKey) { return jsonResponse({ success: false, message: "DUPLICATE_ENTRY" }); } } // SAVE TO SHEET sheet.appendRow([ new Date(), // timestamp data.tripDate, // date data.driver, // driver data.passenger, // passenger data.timeIn, // time in data.timeOut, // time out data.breakfast, // breakfast data.lunch, // lunch data.dinner, // dinner data.parkingFee, // parking fee receiptUrls.join(", "), // receipts links total // total ]); return jsonResponse({ success: true, message: "SAVED" }); } catch (err) { return jsonResponse({ success: false, message: err.toString() }); } } /** * UPLOAD BASE64 IMAGE TO GOOGLE DRIVE */ function uploadBase64ToDrive(base64, name, driver, date) { const folder = DriveApp.getFolderById("17IoUjnNpBivAATe6SMauJ9rLyLr6MGEi"); const blob = Utilities.newBlob( Utilities.base64Decode(base64), "image/jpeg", driver + "_" + date + "_" + name ); const file = folder.createFile(blob); file.setSharing( DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW ); return file.getUrl(); } /** * JSON RESPONSE HELPER */ function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
